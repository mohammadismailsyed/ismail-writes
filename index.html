const express = require("express"); const app = express(); app.use(express.json()); // =============================== // 🌿 FRONTEND (UI + ANIMATION) // =============================== app.get("/", (req, res) => { res.send(` AgriTech AI Platform

🌿 AgriTech AI Platform

🤖 AI Farming Assistant

🌦 Weather → Crop Suggestion

🌿 Crop Disease Detection

`); }); // =============================== // 🤖 AI ASSISTANT API // =============================== app.post("/api/ai", (req,res)=>{ let msg = req.body.message.toLowerCase(); if(msg.includes("soil")) return res.json({reply:"Use compost and organic manure for healthy soil."}); if(msg.includes("pest")) return res.json({reply:"Use neem oil spray and biological pest control."}); if(msg.includes("fertilizer")) return res.json({reply:"Organic fertilizers like cow dung compost are best."}); res.json({reply:"Organic farming improves soil health and sustainability."}); }); // =============================== // 🌦 WEATHER → CROPS // =============================== app.post("/api/weather",(req,res)=>{ let temp = Number(req.body.temp); let crops = temp > 30 ? ["Millets","Cotton"] : temp > 20 ? ["Wheat","Vegetables"] : ["Potato","Peas"]; res.json({crops}); }); // =============================== // 🌿 DISEASE DETECTION // =============================== app.post("/api/disease",(req,res)=>{ let img = req.body.imageName.toLowerCase(); if(img.includes("leaf")) return res.json({ disease:"Leaf Blight", cure:"Use copper-based fungicide and remove infected leaves" }); if(img.includes("spot")) return res.json({ disease:"Leaf Spot", cure:"Spray neem oil and maintain spacing" }); return res.json({ disease:"Healthy Plant", cure:"No treatment required" }); }); // =============================== app.listen(5000,()=>{ console.log("🌿 AgriTech running on http://localhost:5000"); });
